public int search(int[] num,int number){

    //2.循环查找 ?
    while(max>=min){
        //3.计算中间的位置
        int mid=(min+max)/2;
        if(nums[mid] > number){ //比中间值小 继续查找中间值左边的元素
             max=mid-1;
        }else if(nums[mid] < number){//比中间值大 继续查找中间值右边的元素
             min=mid+1;
        }else{ //等于中间值
             return mid;
        }
    }
    return -1;
}
